        function parseDate(dateStr) {
            if(!dateStr) return new Date(0);
            let parts = String(dateStr).split('/');
            if(parts.length === 3) {
                return new Date(parts[2], parts[1] - 1, parts[0]);
            }
            return new Date(dateStr);
        }

        async function fetchLiveData() {
            if (!GOOGLE_SCRIPT_URL) return;
            try {
                const response = await fetch(GOOGLE_SCRIPT_URL);
                const data = await response.json();
                if (data) {
                    database.donations = data.donations || [];
                    database.expenses = data.expenses || [];
                    database.updates = data.updates || [];
                    database.committee = data.committee || [];
                    database.avetiDonations = data.avetiDonations || [];
                    database.avetiExpenses = data.avetiExpenses || [];
                }
                updateDashboard();
            } catch (err) {
                console.error("Error fetching live data:", err);
            }

        function openUpdateModalByIndex(index) {
            let sortedUpdates = [...database.updates].sort((a, b) => parseDate(b.date) - parseDate(a.date));
            let item = sortedUpdates[index];
            if (!item) return;

            document.getElementById('modalDate').innerText = item.date || '';
            document.getElementById('modalTitle').innerText = item.title || '';
            document.getElementById('modalBody').innerText = item.body || '';
            
            const mImg1 = document.getElementById('modalImage1');
            const mImg2 = document.getElementById('modalImage2');
            
            if (item.image1 && item.image1 !== "undefined" && item.image1 !== "") {
                mImg1.src = item.image1;
                mImg1.classList.remove('hidden');
            } else {
                mImg1.classList.add('hidden');
            }
            
            if (item.image2 && item.image2 !== "undefined" && item.image2 !== "") {
                mImg2.src = item.image2;
                mImg2.classList.remove('hidden');
            } else {
                mImg2.classList.add('hidden');
            }

            document.getElementById('updatesModal').classList.remove('hidden');
        }
